736  return v;

741  depth_stack.push(InArray);

743  if (i == std::string::npos)

746  return Value();

756  depth_stack.push(InObject);

762  return Value();

766  v = Value(DeserializeInternal(obj_str, depth_stack));
767  str = str.substr(i + 1, str.length());
768  }
769  else if (str[0] == '\"')

773  if (end_quote == std::string::npos)
774  {

776  return Value();

780  str = str.substr(end_quote + 1, str.length());
781  }
782  else

785  // Numbers can contain an exponent indicator ('e') or a decimal point.
786  bool has_dot = false;

788  std::string temp_val;
789  size_t i = 0;

798  {

802  }
803

805  }

807  {
808  if ((_stricmp(temp_val.c_str(), "fals") != 0) && (_stricmp(temp_val.c_str(), "tru") != 0))

811  if (!found_digit)
812  {
813  // As per JSON standards, a digit must precede the 'e' notation
814  *had_error = true;

816  }

928

936 {
937  Array a;

940  str = Trim(str);
941 
942  // Arrays begin and end with [], so if we don't find one, it's an error
943  if ((str[0] == '[') && (str[str.length() - 1] == ']'))
944  str = str.substr(1, str.length() - 2);
945  else
946  return Value();
947 
948  // extract out all values from the array (remember, a value can also be an array or an object)
949  while (str.length() > 0)
950  {
951  std::string tmp;
952 
953  size_t i = 0;
954  for (; i < str.length(); i++)
955  {
956  // If we get to an object or array, parse it:
957  if ((str[i] == '{') || (str[i] == '['))
958  {
959  Value v = DeserializeValue(str, &had_error, depth_stack);
960  if (had_error)
961  return Value();
962

964  a.push_back(v);
965 
966  break;
967  }
968 
969  bool terminate_parsing = false;
970 
971  if ((str[i] == ',') || (str[i] == ']'))
972  terminate_parsing = true; // hit the end of a value, parse it in the next block
973  else
974  {
975  // keep grabbing chars to build up the value
976  tmp += str[i];
977  if (i == str.length() - 1)
978  terminate_parsing = true; // end of string, finish parsing
979  }
980 
981  if (terminate_parsing)
982  {
983  Value v = DeserializeValue(tmp, &had_error, depth_stack);
984  if (had_error)
985  return Value();
986 
987  if (v.GetType() != NULLVal)
988  a.push_back(v);
989 
990  str = str.substr(i + 1, str.length());
991  break;
992  }
993  }
994  }
